  <h1 id="cardTitle">立即点击</h1>

<script>
  // 这里是自定义变量区，方便替换
  const imageURL = "https://cdn.jsdelivr.net/gh/jackwangs-projects-01/static/blue-icon.png";
  const titleText = "立即点击";
  const descText = "点击链接找到我";
  const linkURL = "https://work.weixin.qq.com/ca/cawcdebb1128a7bc65";

  // 赋值页面元素
  const cardImg = document.getElementById("cardImg");
  const cardTitle = document.getElementById("cardTitle");
  const cardDesc = document.getElementById("cardDesc");
  const cardBtn = document.getElementById("cardBtn");
  const jumpCard = document.getElementById("jumpCard");

  cardImg.src = imageURL;
  cardTitle.textContent = titleText;
  cardDesc.textContent = descText;

  // 点击卡片或按钮都跳转
  function jump() {
    window.location.href = linkURL;
  }
  jumpCard.addEventListener("click", jump);
  cardBtn.addEventListener("click", jump);
  
  // 支持键盘回车跳转（无障碍）
  jumpCard.addEventListener("keydown", (e) => {
    if(e.key === "Enter" || e.key === " ") {
      e.preventDefault();
      jump();
    }
  });
</script>
